# 程序开始
-> program
    out_declare
    out_define
# 外部声明
-> out_declare
    function_declare
# 外部定义
-> out_define
    value_define
    function_define
    array_define
# 函数声明
-> function_declare
    decorate data_type identity ( args_list ) ;
# 变量声明和定义
-> value_define
    decorate data_type identity_list ;
# 函数定义
-> function_define
    decorate data_type identity ( args_list ) { function_content }
# 数组声明和定义
-> array_define
    decorate data_type identity [ value ] Z
# 数组初始化过程
-> Z
    ;
    = { value_list } ;
# 变量组 -
-> value_list
    value
    value, value_list
# 修饰符
-> decorate
    static
    const
# 数据类型
-> data_type
    int
    float
    double
    char
# 参数列表
-> args_list
    decorate data_type identity A
# 多个参数列表辅助
-> A
    null
    , args_list A
# 标示符列表
-> identity_list
    identity B
    identity = expression B
# 辅助
-> B
    null
    , identity B
    , identity = expression B
# 函数体
-> function_content
    statements
# 表达式
-> expression
    identity
    expression + expression
    ( expression )
    expression * expression
    expression / expression
    expression || expression
    expression && expression
# 语句块
-> statements
    value_define
    if_statement
    while_statement
    do_while_statement
    for_statement
    switch_statement
    sentence
# if
-> if_statement
    if ( expression ) statements C
# 辅助if
-> C
    else if ( expression ) statements C
    else statements


-> sentence
    expression ;
    identity assignment expression ;

-> assignment
    =
    +=
    *=
    /=
    %=
